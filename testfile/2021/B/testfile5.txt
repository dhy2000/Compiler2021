int testRecurrence(int n){
	// 变速下降的递归函数 
	if (n==0){
		return 1;
	}
	int temp;  // 递归函数里的临时变量 
	if (n==1){
		temp = 1;
	}else{
		temp = testRecurrence(n-2);
	}
	return temp * testRecurrence(n-1) * n * testRecurrence(n-1);
}

int getSumOfArray(int l,int nums[]){
	int i = 0;
	int sum = 0;
	while(i<l){
		sum = sum + nums[i];
		i = i + 1;
	}
	return (sum + 1 - 1 * 1 / 1) ;
}

int getSumOfArray2(int l,int line,int nums[][3]){
	int sum = 0;
	while(l>0){
		sum = sum + nums[l-1][line];
		l = l - 1;
	}
	return sum ;
}

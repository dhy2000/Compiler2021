int fun3(int a, int b)
{
    if (a > b)
    {
        return a;
    }
    else if (a < b)
    {
        return b;
    }
    return a;
}

int gcd(int a, int b)
{
    if (a % b == 0)
    {
        return b;
    }
    return gcd(b, a % b);
}

int lcm(int a, int b)
{
    int gcds = gcd(a, b);
    return a * b / gcds;
}

int fun5(int a)
{
    if (a == 1)
    {
        return 1;
    }
    else if (a == 2)
    {

    }
    return fun5(a - 1) + fun5(a - 2);
}
